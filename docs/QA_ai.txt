语音接口
// 生成音频并直接播放
async function speak(text, voice = 'en-US-JennyNeural') {
  const res = await fetch('http://localhost:5003/tts', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ text, voice, rate: '+0%', pitch: '+0Hz' })
  });
  const blob = await res.blob();
  const audio = new Audio(URL.createObjectURL(blob));
  audio.play();
}

// 下载为 mp3
async function download(text, filename = 'output.mp3') {
  const res = await fetch('http://localhost:5003/tts', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ text, voice: 'en-US-GuyNeural' })
  });
  const blob = await res.blob();
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = filename;
  a.click();
}

// 获取所有可用声音
async function getVoices() {
  const res = await fetch('http://localhost:5003/voices');
  return res.json();
}
